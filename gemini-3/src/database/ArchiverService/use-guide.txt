# Connected to the archiver service container
docker exec -it archiver_service_container bash

# Created a test file
echo "This is test data for valve 15" > test_file.txt

# Uploaded the file as a test for valve 15
curl -X POST "http://localhost:80/api/tests?valveId=15" \
  -H "Content-Type: multipart/form-data" \
  -F "file=@test_file.txt"

# Retrieved the test metadata
curl -X GET "http://localhost:80/api/tests/1001"

# Downloaded the test file
curl -X GET "http://localhost:80/api/tests/1001/download" --output downloaded_test.txt
